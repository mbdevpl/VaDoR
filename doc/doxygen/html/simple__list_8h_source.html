963  // therefore has two precedeeing elements as its sub-heaps.

980

984  prevprev_heap_index = heaps_len-1-LP[prev_heap];
985  else if(heaps_len > 1)
986  prevprev_heap_index = heaps_len-2;
987  size_t prevprev_heap = heaps_len > 1 ? heaps[prevprev_heap_index] : 0;
988 
989  // create a heap depending on sizes of the last two heaps
990  if(heaps_len >= 2 && prev_heap + 1 == prevprev_heap)
991  heaps[heaps_len] = prev_heap + 2;
992  else if(heaps_len >= 1 && prev_heap == 1)
993  heaps[heaps_len] = 0;
994  else if(heaps_len >= 0)
995  heaps[heaps_len] = 1;
996 
997  // set current heap variable
998  L curr_heap_index = heaps_len;
999  size_t curr_heap = heaps[curr_heap_index];
